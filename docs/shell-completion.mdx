# Shell Completion

Shell completion is available for `bash`, `zsh`, `fish` and `powershell`.

import Tabs from '@theme/Tabs';
import TabItem from '@theme/TabItem';

<Tabs
    defaultValue="bash"
    values={[
        { label: 'bash', value: 'bash', },
        { label: 'zsh', value: 'zsh', },
        { label: 'fish', value: 'fish', },
        { label: 'powershell', value: 'powershell', },
    ]}
>

<TabItem value="bash">

```bash
repoctl completion bash
```

</TabItem>

<TabItem value='zsh'>

```bash
repoctl completion zsh
```

</TabItem>

<TabItem value='fish'>

```bash
repoctl completion fish
```

</TabItem>

<TabItem value='powershell'>

```bash
repoctl completion powershell
```

</TabItem>

</Tabs>
